# DarajaExpressResponse

Payment Response

## Example Usage

```typescript
import { DarajaExpressResponse } from "@moflay/sdk/models";

let value: DarajaExpressResponse = {
  paymentId: "pay_ABC123DEF456GHI",
  transactionId: "trxn_ABC123DEF456GHI",
  customerId: "cus_GqfKXLmg61LURZhB",
  phoneNumber: "254712345678",
  amount: 1000,
  currency: "KES",
  status: "pending",
  description: "Premium plan",
  accountReference: "ORDER123",
  environment: "sandbox",
  timestamp: new Date("2024-12-19T10:25:00.000Z"),
  message: "Payment request sent to customer",
};
```
